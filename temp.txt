#include "BaseCamera.h"

CBaseCamera::CBaseCamera( bool bRenderToTexture, unsigned int uiMSAALevel, const glm::ivec2 &vecSize, float flFOV, float flNear, float flFar, const glm::vec3 &vecPosition, const glm::vec3 &vecRotation, const glm::vec3 &vecScale, bool bShouldDraw, bool bActive ) : BaseClass( vecPosition, vecRotation, vecScale, bShouldDraw, bActive )
{
	m_bRenderToTexture = bRenderToTexture;

	m_uiMSAALevel = uiMSAALevel;
	m_vecSize = vecSize;
	m_flFOV = flFOV;
	m_flNear = flNear;
	m_flFar = flFar;
	m_bUpdateProjection = false;

	UpdateProjection();
	UpdateView();

	m_bProjectionUpdated = true;
	m_bViewUpdated = true;

	if (m_uiMSAALevel != 0)
		CreateMSAABuffers();

	if (m_bRenderToTexture)
		CreateTextureBuffers();
}

void CBaseCamera::PostThink( void )
{
	if (m_bUpdateProjection)
	{
		UpdateProjection();
		m_bUpdateProjection = false;
		m_bProjectionUpdated = true;
	}
	else
	{
		m_bProjectionUpdated = false;
	}

	if (HasTransformUpdated())
	{
		UpdateView();
		ResetTransformUpdated();
		m_bViewUpdated = true;
	}
	else
	{
		m_bViewUpdated = false;
	}
}

void CBaseCamera::SetMSAALevel( unsigned int uiMSAALevel )
{
	if (m_uiMSAALevel != uiMSAALevel)
	{
		m_uiMSAALevel = uiMSAALevel;

		if (m_uiMSAALevel != 0)
		{
			DestroyMSAABuffers();
			CreateMSAABuffers();
		}
		else
		{
			DestroyMSAABuffers();
		}
	}
}

void CBaseCamera::SetSize( const glm::ivec2 &vecSize )
{
	if (m_vecSize != vecSize)
	{
		if ((float)m_vecSize.x / (float)m_vecSize.y != (float)vecSize.x / (float)vecSize.y)
			m_bUpdateProjection = true;

		m_vecSize = vecSize;

		if (m_uiMSAALevel != 0)
		{
			DestroyMSAABuffers();
			CreateMSAABuffers();
		}

		if (m_bRenderToTexture)
		{
			DestroyTextureBuffers();
			CreateTextureBuffers();
		}
	}
}

void CBaseCamera::SetFOV( float flFOV )
{
	if (m_flFOV != flFOV)
	{
		m_flFOV = flFOV;
		m_bUpdateProjection = true;
	}
}

void CBaseCamera::SetNear( float flNear )
{
	if (m_flNear != flNear)
	{
		m_flNear = flNear;
		m_bUpdateProjection = true;
	}
}

void CBaseCamera::SetFar( float flFar )
{
	if (m_flFar != flFar)
	{
		m_flFar = flFar;
		m_bUpdateProjection = true;
	}
}

bool CBaseCamera::ProjectionUpdated( void ) const
{
	return m_bProjectionUpdated;
}

bool CBaseCamera::ViewUpdated( void ) const
{
	return m_bViewUpdated;
}

const glm::mat4 &CBaseCamera::GetProjection( void ) const
{
	return m_matProjection;
}

const glm::mat4 &CBaseCamera::GetView( void ) const
{
	return m_matView;
}

const glm::vec3 &CBaseCamera::GetFront( void ) const
{
	return m_vecFront;
}

const glm::vec3 &CBaseCamera::GetUp( void ) const
{
	return m_vecUp;
}

const glm::vec3 &CBaseCamera::GetRight( void ) const
{
	return m_vecRight;
}

void CBaseCamera::UpdateProjection( void )
{
	m_matProjection = glm::perspective( m_flFOV, (float)m_vecSize.x / (float)m_vecSize.y, m_flNear, m_flFar );
}

void CBaseCamera::UpdateView( void )
{
	m_vecFront = GetRotation() * g_vecFront;
	m_vecUp = GetRotation() * g_vecUp;
	m_vecRight = GetRotation() * g_vecRight;

	m_matView = glm::lookAt( GetPosition(), GetPosition() + m_vecFront, m_vecUp );
}

void CBaseCamera::CreateMSAABuffers( void )
{
	glGenFramebuffers( 1, &m_uiMSAAFBO );
	glGenTextures( 1, &m_uiMSAA );
	glGenRenderbuffers( 1, &m_uiMSAARBO );

	glBindFramebuffer( GL_FRAMEBUFFER, m_uiMSAAFBO );
	glBindTexture( GL_TEXTURE_2D_MULTISAMPLE, m_uiMSAA );
	glBindRenderbuffer( GL_RENDERBUFFER, m_uiMSAARBO );

	glTexImage2DMultisample( GL_TEXTURE_2D_MULTISAMPLE, m_uiMSAALevel, GL_RGB8, m_vecSize.x, m_vecSize.y, GL_TRUE );
	glFramebufferTexture2D( GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D_MULTISAMPLE, m_uiMSAA, 0 );
	glRenderbufferStorageMultisample( GL_RENDERBUFFER, m_uiMSAALevel, GL_DEPTH24_STENCIL8, m_vecSize.x, m_vecSize.y );
	glFramebufferRenderbuffer( GL_FRAMEBUFFER, GL_DEPTH_STENCIL_ATTACHMENT, GL_RENDERBUFFER, m_uiMSAARBO );
}

void CBaseCamera::DestroyMSAABuffers( void )
{
	glDeleteFramebuffers( 1, &m_uiMSAAFBO );
	glDeleteTextures( 1, &m_uiMSAA );
	glDeleteRenderbuffers( 1, &m_uiMSAARBO );
}

void CBaseCamera::CreateTextureBuffers( void )
{
	glGenFramebuffers( 1, &m_uiTextureFBO );
	glGenTextures( 1, &m_uiTexture );

	glBindTexture( GL_TEXTURE_2D, m_uiTexture );
	glTexImage2D( GL_TEXTURE_2D, 0, GL_RGB8, m_vecSize.x, m_vecSize.y, 0, GL_RGB, GL_UNSIGNED_BYTE, NULL );
	glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR );
	glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR );
	glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT );
	glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT );
	glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_COMPARE_MODE, GL_COMPARE_REF_TO_TEXTURE );
	glBindFramebuffer( GL_FRAMEBUFFER, m_uiTextureFBO );
	glFramebufferTexture2D( GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, m_uiTexture, 0 );

	// glDrawBuffer( GL_NONE ); // See what this does exactly
	// glReadBuffer( GL_NONE ); // See what this does exactly
}

void CBaseCamera::DestroyTextureBuffers( void )
{
	glDeleteFramebuffers( 1, &m_uiTextureFBO );
	glDeleteTextures( 1, &m_uiTexture );
}
